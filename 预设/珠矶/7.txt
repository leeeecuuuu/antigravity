<script>
    // Ensure script runs after DOM is loaded
    document.addEventListener('DOMContentLoaded', () => {
        const multiPerspectiveContainer = document.querySelector('.multi-perspective-container'); // Check if the perspective section exists

        // Only run tab logic if the perspective container is present
        if (multiPerspectiveContainer) {
            const tabsContainer = multiPerspectiveContainer.querySelector('.perspective-tabs');
            const panelsContainer = multiPerspectiveContainer.querySelector('.perspective-panels');

            // Check if tabs and panels exist within the container
            if (tabsContainer && panelsContainer) {
                const tabs = tabsContainer.querySelectorAll('button[role="tab"]');
                const panels = panelsContainer.querySelectorAll('.perspective-panel');

                if (tabs.length > 0 && panels.length > 0) { // Ensure tabs/panels were found

                    // Function to switch tabs
                    function switchTab(targetTab) {
                        if (!targetTab) return; // Exit if targetTab is null

                        const targetPanelId = targetTab.getAttribute('aria-controls');
                        const targetPanel = panelsContainer.querySelector(`#${targetPanelId}`);

                        // Deactivate all tabs
                        tabs.forEach(t => {
                            t.classList.remove('active');
                            t.setAttribute('aria-selected', 'false');
                        });

                        // Activate the target tab
                        targetTab.classList.add('active');
                        targetTab.setAttribute('aria-selected', 'true');

                        // Scroll the panels container to the target panel
                        if (targetPanel) {
                            const scrollLeft = targetPanel.offsetLeft;
                            // Check if scrollLeft is a valid number before scrolling
                            if (typeof scrollLeft === 'number') {
                                panelsContainer.scrollTo({
                                    left: scrollLeft,
                                    behavior: 'smooth'
                                });
                            }
                        }
                    }

                    // Add click listener to tabs container
                    tabsContainer.addEventListener('click', (e) => {
                        if (e.target.matches('button[role="tab"]')) {
                            switchTab(e.target);
                        }
                    });

                    // Optional: Update tabs when user scrolls/swipes the panels
                    let scrollTimeout;
                    panelsContainer.addEventListener('scroll', () => {
                        clearTimeout(scrollTimeout);
                        scrollTimeout = setTimeout(() => {
                            const containerWidth = panelsContainer.offsetWidth;
                            if (containerWidth > 0) { // Avoid division by zero
                                const scrollLeft = panelsContainer.scrollLeft;
                                const activeIndex = Math.round(scrollLeft / containerWidth);

                                if (activeIndex >= 0 && activeIndex < tabs.length) {
                                    tabs.forEach((t, index) => {
                                        if (index === activeIndex) {
                                            if (!t.classList.contains('active')) {
                                                t.classList.add('active');
                                                t.setAttribute('aria-selected', 'true');
                                            }
                                        } else {
                                            if (t.classList.contains('active')) {
                                                t.classList.remove('active');
                                                t.setAttribute('aria-selected', 'false');
                                            }
                                        }
                                    });
                                }
                            }
                        }, 150);
                    });

                    // Initial state check (optional if first tab always has 'active' class in HTML)
                    // if (tabs.length > 0) {
                    //     switchTab(tabs[0]);
                    // }
                }
            }
        }
    }); // End DOMContentLoaded
</script>

        let isCgActive = false;
        let gameStates = {};

        // --- 脚本解析正则表达式 ---
        const scriptRegex = {
            background: /^(背景|CG)\|(.+)$/,
            dialogue: /^(.+?)\|(.+?)\|(.+)$/,
            narrator: /^(.+?)\|旁白$/
        };

        // --- 核心功能 ---
        function nextLine() {
            if (isTyping) {
                // 如果正在打字,立即显示完整文本
                isTyping = false;
                return;
            }

            if (currentLineIndex >= scriptLines.length - 1) {
                // 到达最后一行,隐藏继续提示,但保留最后的对话内容
                continueIndicator.style.display = 'none';
                return;
            }

            currentLineIndex++;
            executeLine(scriptLines[currentLineIndex]);
        }

        function executeLine(line) {
            console.log('执行:', line);
            
            // 优先匹配背景指令
            const backgroundMatch = line.match(scriptRegex.background);
            if (backgroundMatch) {
                handleBackgroundCommand(backgroundMatch);
                // 背景或CG指令后,自动处理下一行
                nextLine();
                return;
            }

            // 匹配旁白
            const narratorMatch = line.match(scriptRegex.narrator);
            if (narratorMatch) {
                handleNarratorCommand(narratorMatch);
                return;
            }

            // 匹配对话
            const dialogueMatch = line.match(scriptRegex.dialogue);
            if (dialogueMatch) {
                handleDialogueCommand(dialogueMatch);
                return;
            }

            // 无法解析的行（包括占位符）,警告并自动跳到下一行
            console.warn('无法解析的脚本行:', line);
            nextLine();
        }
        
        function typeWriter(element, text, onComplete) {
            isTyping = true;
            let i = 0;
            element.innerHTML = '';
            
            function type() {
                if (!isTyping) { // 如果中途停止打字
                    element.innerHTML = text;
                    if(onComplete) onComplete();
                    return;
                }
                
                if (i < text.length) {
                    element.innerHTML += text.charAt(i);
                    i++;
                    setTimeout(type, 50); // 打字速度
                } else {
                    isTyping = false;
                    if(onComplete) onComplete();
                }
            }
            type();
        }

        // --- 指令处理器 ---
        function handleBackgroundCommand(match) {
            const [, type, content] = match;
            const bgKey = Object.keys(imageDatabase.backgrounds).find(k => k === content.trim());
            if (bgKey) {
                const data = imageDatabase.backgrounds[bgKey];
                if (type.toUpperCase() === 'CG') {
                    isCgActive = true;
                    document.body.style.backgroundImage = `url('${data.url}')`;
                    hideCharacter();
                    statusBar.classList.add('hidden');
                } else {
                    isCgActive = false;
                    document.body.style.backgroundImage = `linear-gradient(rgba(0,0,0,0.4), rgba(0,0,0,0.6)), url('${data.url}')`;
                    statusBar.classList.remove('hidden');
                }
            }
        }

        function handleDialogueCommand(match) {
            const [, dialogue, speaker, imageKeyword] = match;
            speakerName.textContent = speaker.toUpperCase();
            speakerName.classList.remove('narrator');

            if (isCgActive) {
                hideCharacter();
            } else {
                const charKey = Object.keys(imageDatabase.characters).find(k => k === imageKeyword.trim());
                if (charKey) {
                    showCharacter(imageDatabase.characters[charKey]);
                } else {
                    // 对话中如果没有指定图片,则保持当前立绘状态,不隐藏
                    // hideCharacter(); 
                }
            }
            
            typeWriter(dialogueText, dialogue);
        }

        function handleNarratorCommand(match) {
            const [, text] = match;
            speakerName.textContent = '旁白';
            speakerName.classList.add('narrator');
            
            if (isCgActive) {
                hideCharacter();
            }
            // 旁白时不改变立绘状态
            typeWriter(dialogueText, text);
        }

        // --- 视图更新 ---
        function showCharacter(data) {
            if (characterImage.src !== data.url) {
                characterImage.src = data.url;
            }
            characterImage.classList.remove('hidden');
        }

        function hideCharacter() {
            characterImage.classList.add('hidden');
        }

        // --- 初始化 ---
        document.addEventListener('DOMContentLoaded', function() {
            // 解析游戏数据
            const statesMatch = gameData.match(/<states>([\s\S]*?)<\/states>/);
            const messageMatch = gameData.match(/<message>([\s\S]*?)<\/message>/);

            if (statesMatch && statesMatch[1]) {
                const statesRaw = statesMatch[1].trim();
                statesRaw.split('\n').forEach(line => {
                    const parts = line.split(':');
                    if (parts.length === 2) {
                        gameStates[parts[0].trim()] = parts[1].trim();
                    }
                });
                updateStatusBar();
            }

            const storyScript = (messageMatch && messageMatch[1]) ? messageMatch[1].trim() : '';

            // 解析脚本
            scriptLines = storyScript.split('\n').map(line => line.trim()).filter(line => line.length > 0);
            
            if (scriptLines.length > 0) {
                // 自动开始,直接跳过第一部分
                setTimeout(() => {
                    nextLine();
                }, 100); // 稍微延迟以确保页面完全加载
            } else {
                dialogueText.innerHTML = '错误：未加载任何脚本内容。';
            }
            
            // 键盘事件监听
            document.addEventListener('keydown', function(e) {
                if (e.key === 'ArrowRight' || e.key === ' ') {
                    e.preventDefault();
                    nextLine();
                }
            });
        });

        function updateStatusBar() {
            if (!statusBar) return;
            let html = '';
            for (const key in gameStates) {
                if (key !== '星辞好感度') {
                    html += `<div class="status-item">${key}: ${gameStates[key]}</div>`;
                }
            }
            statusBar.innerHTML = html;
        }

        characterImage.addEventListener('error', function() {
            console.error('角色图片加载失败:', this.src);
            hideCharacter();
        });
    </script>
</body>
