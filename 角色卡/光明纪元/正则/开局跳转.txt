      console.log(`[UI] 用户选择开场 ID: ${openingId}`);
      const zeroBasedSwipeIndex = openingId;
      console.log(`[API 准备] 映射的目标 Swipe 索引 (0-based): ${zeroBasedSwipeIndex}`);

      try {
        if (typeof getChatMessages === 'function' && typeof setChatMessage === 'function') {
          console.log('[API] 检测到 SillyTavern 函数。');
          const messages = await getChatMessages(0, { include_swipe: true });
          const firstMessage = messages?.[0];
          const swipes = firstMessage?.swipes;
          const availableCount = swipes?.length ?? 0;

          if (swipes && availableCount > zeroBasedSwipeIndex) {
            const targetSwipeContent = swipes[zeroBasedSwipeIndex];
            console.log(`[API] 在索引 ${zeroBasedSwipeIndex} 找到内容, 正在切换...`);
            await setChatMessage(targetSwipeContent, 0, {
              swipe_id: zeroBasedSwipeIndex,
              refresh: 'display_and_render_current'
            });

            console.log(`[成功] 已切换至: ${title}`);

    // 更新页面内容
    function updatePages() {
      if (currentPage < pageContents.length) {
        leftContent.innerHTML = pageContents[currentPage].left;
        rightContent.innerHTML = pageContents[currentPage].right;
        leftPageNum.textContent = currentPage * 2 + 1;
        rightPageNum.textContent = currentPage * 2 + 2;

        // 添加翻页动画
        leftContent.parentElement.classList.add('page-transition');
        rightContent.parentElement.classList.add('page-transition');

        setTimeout(() => {
          leftContent.parentElement.classList.remove('page-transition');
          rightContent.parentElement.classList.remove('page-transition');
        }, 600);
      }

      // 控制按钮显示
      prevBtn.style.display = currentPage > 0 ? 'flex' : 'none';
      nextBtn.style.display = currentPage < pageContents.length - 1 ? 'flex' : 'none';
    }

    // 上一页
    prevBtn.addEventListener('click', function() {
      if (currentPage > 0) {
        currentPage--;
        updatePages();
      }
    });

    // 下一页
    nextBtn.addEventListener('click', function() {
      if (currentPage < pageContents.length - 1) {
        currentPage++;
        updatePages();
      }
    });

    // 键盘控制
    document.addEventListener('keydown', function(e) {
      if (bookOpen.classList.contains('visible')) {
        if (e.key === 'ArrowLeft' && currentPage > 0) {
          currentPage--;
          updatePages();
        } else if (e.key === 'ArrowRight' && currentPage < pageContents.length - 1) {
          currentPage++;
          updatePages();
        }
      }
    });
