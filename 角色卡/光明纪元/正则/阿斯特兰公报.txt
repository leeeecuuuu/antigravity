    if (window.jQuery) {
      $(() => {
        function initTabSwitching(container) {
          container.find('.tab-btn').on('click', function () {
            const $btn = $(this);
            const targetTab = $btn.data('tab');
            const $contentArea = $btn.closest('.gazette-content');
            $contentArea.find('.tab-btn').removeClass('active');
            $btn.addClass('active');
            $contentArea.find('.tab-content').removeClass('active');
            $contentArea.find(`.tab-content[data-content="${targetTab}"]`).addClass('active');
          });
        }
        function parseMissionData(container) {
            const $log = container.find('.mission-log-container');
            const missionData = $log.data('mission');
            $log.empty();
            if (!missionData || typeof missionData !== 'string' || missionData.startsWith('$')) {
                $log.html('<div class="list-empty-message">暂无任务记录</div>');
                return;
            }
            const missions = missionData.split(';').map(m => m.trim()).filter(m => m);
            if (missions.length === 0) {
                $log.html('<div class="list-empty-message">暂无任务记录</div>');
                return;
            }
            missions.forEach(missionStr => {
                const [typePart, contentPart] = missionStr.split(/:(.*)/s);
                if (!contentPart) return;
                const type = typePart.trim();
                const [title, desc] = contentPart.split('|').map(s => s.trim());
                const $item = $('<div>').addClass('mission-item');
                const $titleDiv = $('<div>').addClass('mission-title').text(title);
                const $typeSpan = $('<span>').addClass('mission-type');
                if (type === '主线') {
                    $item.addClass('main-quest');
                    $typeSpan.text('主线');
                } else if (type === '支线') {
                    $typeSpan.text('支线');
                } else if (type === '感情线') {
                    $typeSpan.text('感情线');
                }
                $titleDiv.prepend($typeSpan);
                $item.append($titleDiv);
                if (desc) {
                    const $descDiv = $('<div>').addClass('mission-desc').text(desc);
                    $item.append($descDiv);
                }
                $log.append($item);
            });
        }
        function parseNewsData(container) {
            const $list = container.find('.news-list');
            const newsData = $list.data('news');
            $list.empty();
            if (!newsData || typeof newsData !== 'string' || newsData.startsWith('$')) {
                $list.html('<div class="list-empty-message">暂无最新情报</div>');
                return;
            }
            const newsItems = newsData.split(';').map(item => item.trim()).filter(item => item);
            if (newsItems.length === 0) {
                $list.html('<div class="list-empty-message">暂无最新情报</div>');
                return;
            }
            newsItems.forEach(item => {
                const match = item.match(/^\[([^\]]+)\]\s*(.*)$/);
                const source = match ? match[1] : '未知来源';
                const text = match ? match[2] : item;
                const $item = $('<div>').addClass('news-item');
                const $text = $('<span>').addClass('news-text').text(text);
                const $source = $('<span>').addClass('news-source').text(`— ${source}`);
                $item.append($text).append($source);
                $list.append($item);
            });
        }
        function parseMapData(container) {
            const $map = container.find('.map-container');
            const mapData = $map.data('map');
            const currentPlace = $map.data('current-place') || '';
            $map.empty();
            if (!mapData || typeof mapData !== 'string' || mapData.startsWith('$')) {
                $map.html('<div class="map-empty-message">无可用地图信息</div>');
                return;
            }
            const mainLocations = mapData.split('[').slice(1);
            if(mainLocations.length === 0) {
                $map.html('<div class="map-empty-message">无可用地图信息</div>');
                return;
            }
            mainLocations.forEach(locString => {
                const parts = locString.split(']');
                const mainName = parts[0];
                const subData = parts[1] || '';
                const $group = $('<div>').addClass('map-location-group');
                const $mainDiv = $('<div>').addClass('map-main-location').text(mainName);
                $group.append($mainDiv);
                const $nodesContainer = $('<div>').addClass('map-nodes-container');
                const subLocations = subData.split('|').map(s => s.trim()).filter(s => s);
                subLocations.forEach(subStr => {
                    const nameMatch = subStr.match(/^[^@#]*/);
                    const subName = nameMatch ? nameMatch[0] : '未知地点';
                    const charMatch = subStr.match(/@([^#]*)/);
                    const characters = charMatch ? charMatch[1].split(',').map(c => c.trim()).filter(Boolean) : [];
                    const eventMatch = subStr.match(/#([^@]*)/);
                    const events = eventMatch ? eventMatch[1].split(',').map(e => e.trim()).filter(Boolean) : [];
                    const $subDiv = $('<div>').addClass('map-sub-location');
                    if (currentPlace && currentPlace.includes(subName) && !currentPlace.startsWith('$')) {
                        $subDiv.addClass('current');
                    }
                    const $nameDiv = $('<div>').addClass('map-sub-name').text(subName);
                    $subDiv.append($nameDiv);
                    if (characters.length > 0 || events.length > 0) {
                        const $tagsContainer = $('<div>').addClass('map-tags-container');
                        characters.forEach(char => {
                            $tagsContainer.append($('<span>').addClass('map-tag character').text(char));
                        });
                        events.forEach(event => {
                            $tagsContainer.append($('<span>').addClass('map-tag event').text(event));
                        });
                        $subDiv.append($tagsContainer);
                    }
                    $nodesContainer.append($subDiv);
                });
                $group.append($nodesContainer);
                $map.append($group);
            });
        }
        function initializeGazette(container) {
            initTabSwitching(container);
            parseMissionData(container);
            parseNewsData(container);
            parseMapData(container);
        }
        $('.gazette-container').each(function() {
            initializeGazette($(this));
        });
      });
    }
  </script>
